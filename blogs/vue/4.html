    Vue+ts+Element plus 封装表单🔥

    vkuyxu 2022/4/30 Vuetypescriptelement plus

    # 在 components 下面创建 myForm.vue 文件

    <template>
  <!-- 插槽 -->
  <div>
    <slot name="header"></slot>
  </div>
  <!-- 动态生成input和select表单 -->
  <el-form :label-width="labelWidth" :model="modelValue" ref="formRef">
    <!-- 不要直接循环formItem -->
    <template v-for="item in formItems" :key="item.user">
      <!-- 输入框 -->
      <el-form-item :label="item.label" :prop="item.name">
        <template v-if="item.type === 'input' || item.type === 'password'">
          <el-input
            :placeholder="item.placeholder"
            v-model="modelValue![item.name]"
            :show-password="item.type === 'password'"
          />
        </template>
        <!-- 下拉框 -->
        <template v-else-if="item.type === 'select'">
          <el-select
            :placeholder="item.placeholder"
            style="width: 240px"
            v-model="modelValue![item.name]"
          >
            <el-option
              v-for="option in item.options"
              :key="option.value"
              :label="option.label"
              :value="option.value"
            />
          </el-select>
        </template>
        <!-- 单选日期框 -->
        <template v-else-if="item.type === 'date'">
          <el-date-picker
            :type="item.type"
            :placeholder="item.placeholder"
            v-model="modelValue![item.name]"
          />
        </template>
        <!-- 多选日期框 -->
        <template v-else-if="item.type === 'datepicker'">
          <el-date-picker
            :type="item.option!.type"
            :startPlaceholder="item.option!.startPlaceholder"
            :endPlaceholder="item.option!.endPlaceholder"
            v-model="modelValue![item.name]"
          />
        </template>
        <!-- 开关 -->
        <template v-else-if="item.type === 'switch'">
          <el-switch v-model="modelValue![item.name]" />
        </template>
      </el-form-item>
    </template>
    <div>
      <slot name="footer"></slot>
    </div>
  </el-form>
</template>

<script setup lang="ts">
import type { FormInstance, FormRules } from "element-plus";
import { ref } from "vue";
const props = defineProps({
  formItems: {
    type: Array<any>,
  },
  labelWidth: {
    type: String,
    default: "120px",
  },
  modelValue: {
    type: Object,
  },
});
// 创建form表单ref实例
const formRef = ref<FormInstance>();
// 默认重置方法
const resetForm = () => {
  formRef.value?.resetFields();
};
// 暴露方法
defineExpose({
  resetForm,
});
    // ！！！ 注意事项
    // 基本上就是通过if else 判断配置项的type属性来决定渲染什么组件
    //记得绑定Ref到form表单上
    // 定义的方法记得暴露出去
</script>

    # 父组件

    <template>
  <myForm v-bind="formConfig" v-model="data" ref="formRef">
    <template #header>
      <div>登录</div>
    </template>
    <template #footer>
      <el-form-item>
        <el-button type="primary" @click="onSubmit">登录</el-button>
        <el-button @click="reset">重置</el-button>
      </el-form-item>
    </template>
  </myForm>
</template>

<script setup lang="ts">
import { ref } from "vue";
import type { FormInstance } from "element-plus";
import myForm from "@/components/myForm.vue";
import formConfig from "@/config/formConfig";
// 封装表单 通过数据处理动态生成表单项
// 表单配置项
// 初始化数据
// 1.取出formItems
const formItems = formConfig.formItems ?? []; // 空值合并操作符  "null" ?? "hi" 返回hi
// 2.创建数据对象
const data = ref<any>({});
formItems.forEach((value) => {
  // 注意数据格式 不同的表单v-model绑定的值不一样
  if (value.type === "switch") {
    data.value[value.type] = false;
  } else {
    data.value[value.type] = "";
  }
});
type resetFormFunc = () => void;
// 创建form表单ref实例
const formRef = ref<FormInstance & { resetForm: resetFormFunc }>();
// 登录方法
const onSubmit = () => {
  console.log(data.value);
};

// 重置方法
const reset = () => {
  formRef.value?.resetForm();
};
// ！！！注意事项
// 表单需要的数据在父组件中定义 用响应式变量
// 通过v-model给子组件绑定数据  子组件通过默认的modelValue进行接受  或者通过
// const model = defineModel() 进行接受
// 这里也要绑定ref
</script>
